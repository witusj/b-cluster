rawDF <- read.csv2(data, stringsAsFactors = FALSE)
cols <- rawDF[1,]
names(rawDF) <- cols
rawDF <- rawDF[-1,]
rawDF[,-1] <- lapply(rawDF[,-1], as.factor)
